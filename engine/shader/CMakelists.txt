    "graphics_shader_generater"
    "${SHADERS}"
    "${CMAKE_CURRENT_SOURCE_DIR}/generated"
    "${CMAKE_CURRENT_SOURCE_DIR}/compiler/DirectX/dxc/dxc.exe")
